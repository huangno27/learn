def gethtml(url):
    """
    获取整个页面的内容
    :param url:
    :return:
    """
    txt2=urllib.request.urlopen(url)
    html = txt2.read()
    return html

def get_title(html):
    """
    获取标题
    :param html:
    :return:
    """
    #标题  那标题
    title_re=re.compile('<h1>(.*?)</h1>')
    title=re.search(title_re,html.decode('gbk'))
    title=title.group(1).replace('\\','').replace('/','').replace(':','').replace('*','').replace('?','').replace('"','').replace('>','').replace('<','').replace('|','').replace(' ','')
    print(title)
    return title
